### 计时器各种场景

```jsx harmony
export default function App() {
  const [seconds, setSeconds] = useState(0);

  useEffect(() => {
    const id = setInterval(() => {
      setSeconds(seconds + 1);
    }, 1000);
  });
  return (
    <div className="App">
      <h1>{Math.floor(seconds)}</h1>
      <h2>Start editing to see some magic happen!</h2>
    </div>
  );
}
```
